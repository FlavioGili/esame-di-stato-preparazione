    <script>

        // Funzione per gestire la selezione della materia

        function selectSubject(subject) {

            // Simulazione di caricamento dei documenti (sostituisci con la tua logica)

            let documents = getDocumentsBySubject(subject);


            // Visualizza i documenti nella pagina

            let documentContainer = document.getElementById('document-container');

            documentContainer.innerHTML = '';

            if (documents.length > 0) {

                documents.forEach(document => {

                    let documentElement = document.createElement('div');

                    documentElement.classList.add('document');

                    documentElement.textContent = document;

                    documentContainer.appendChild(documentElement);

                });

            } else {

                documentContainer.innerHTML = '<p>Nessun documento disponibile per questa materia.</p>';

            }

        }


        // Funzione di esempio per ottenere i documenti in base alla materia (sostituisci con la tua logica)

        function getDocumentsBySubject(subject) {

            // Simulazione di dati di esempio

            switch (subject) {

                case 'Italiano':

                    return ['Documento 1', 'Documento 2', 'Documento 3'];

                case 'Matematica':

                    return ['Documento 4', 'Documento 5'];

                case 'Storia':

                    return ['Documento 6'];

                default:

                    return [];

            }

        }

    </script>
